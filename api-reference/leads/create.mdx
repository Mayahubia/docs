---
title: "Criar lead"
api: "POST /user/lead"
description: "Criar um novo lead no sistema Maya Hub
"
---

Este endpoint permite criar um novo lead no sistema Maya Hub.

### Corpo da solicitação

<ParamField body="phone_number" type="string" required>
  O número de telefone do lead no formato E.164 (exemplo: +1234567890)
</ParamField>

<ParamField body="campaign_id" type="integer" required>
  O ID da campanha para a qual o lead será criado
</ParamField>

<ParamField body="variables" type="object">
  As variáveis a serem passadas para o lead

  <Expandable title="Propriedades das Variaveis">
    <ParamField body="customer_name" default="<string>" type="string">
      O nome do cliente
    </ParamField>
    <ParamField body="email" default="<string>" type="string">
      O email do cliente
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="allow_dupplicate" type="boolean">
  Se deve permitir leads duplicados em uma campanha.
</ParamField>

<ParamField body="secondary_contacts" type="array">
  Array de contatos secundários a serem criados junto com o lead principal

  <Expandable title="Propriedades de secondary_contacts">
    <ParamField body="phone_number" type="string" required>
      The phone number of the secondary contact in E.164 format
    </ParamField>
    <ParamField body="variables" type="object">
      The variables to pass to the secondary contact

      <Expandable title="variables properties">
        <ParamField body="customer_name" default="<string>" type="string">
          The name of the secondary contact
        </ParamField>
        <ParamField body="email" default="<string>" type="string">
          The email of the secondary contact
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

### Respostas

<ResponseField name="message" type="string" default="Lead created successfully">
  A mensagem da resposta
</ResponseField>

<ResponseField name="data" type="object">
  O objeto de dados do lead

  <Expandable title="Propriedades de data">
    <ResponseField name="id" type="integer">
      O ID exclusivo do lead criado
    </ResponseField>
    <ResponseField name="campaign_id" type="integer">
      O ID da campanha à qual este lead pertence
    </ResponseField>
    <ResponseField name="phone_number" type="string">
      O número de telefone do lead no formato E.164
    </ResponseField>
    <ResponseField name="variables" type="object">
      As variáveis associadas ao lead

      <Expandable title="Propriedades das Variáveis">
        <ResponseField name="customer_name" type="string" default="<string>">
          O nome do cliente
        </ResponseField>
        <ResponseField name="email" type="string" default="<string>">
          O email do cliente
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="status" type="string" default="created">
      O status do lead
    </ResponseField>
    <ResponseField name="created_at" type="string">
      O carimbo de data e hora em que o lead foi criado
    </ResponseField>
    <ResponseField name="updated_at" type="string">
      O carimbo de data e hora da última atualização do lead
    </ResponseField>
    <ResponseField name="campaign" type="object">
      As informações da campanha

      <Expandable title="Propriedades da Campanha">
        <ResponseField name="id" type="integer">
          O ID da campanha
        </ResponseField>
        <ResponseField name="name" type="string">
          O nome da campanha
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="secondary_contacts" type="array">
      Array de contatos secundários associados a este lead

      <Expandable title="Propriedades de secondary_contacts">
        <ResponseField name="id" type="integer">
          O ID do lead de contato secundário
        </ResponseField>
        <ResponseField name="phone_number" type="string">
          O número de telefone do contato secundário no formato E.164
        </ResponseField>
        <ResponseField name="variables" type="object">
          As variáveis associadas ao contato secundário
        </ResponseField>
        <ResponseField name="status" type="string">
          O status do contato secundário
        </ResponseField>
        <ResponseField name="created_at" type="string">
          A data de criação do contato secundário
        </ResponseField>
        <ResponseField name="updated_at" type="string">
          A data de atualização do contato secundário
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
