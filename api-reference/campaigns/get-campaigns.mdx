---
title: "Listar campanhas"
api: "GET /user/campaigns"
description: "Listar todas as Campanhas"
---

Este endpoint permite listar todas as campanhas.

### Campos da resposta

<ResponseField name="campaigns" type="array">
  <Expandable title="Propriedades">
    <ResponseField name="id" type="integer">
      O ID da campanha
    </ResponseField>
    <ResponseField name="name" type="string">
      O nome da campanha
    </ResponseField>
    <ResponseField name="status" type="string">
      O status da campanha
    </ResponseField>
    <ResponseField name="max_calls_in_parallel" type="integer">
      Número máximo de chamadas que podem ser realizadas em paralelo
    </ResponseField>
    <ResponseField name="mark_complete_when_no_leads" type="boolean">
      Se a campanha deve ser marcada como concluída quando não houver leads
    </ResponseField>
    <ResponseField name="allowed_hours_start_time" type="string">
      O horário de início para as horas permitidas de chamadas
    </ResponseField>
    <ResponseField name="allowed_hours_end_time" type="string">
      O horário de término para as horas permitidas de chamadas
    </ResponseField>
    <ResponseField name="allowed_days" type="array">
      Os dias da semana em que as chamadas são permitidas
    </ResponseField>
    <ResponseField name="max_retries" type="integer">
      Número máximo de tentativas de repetição para chamadas falhadas
    </ResponseField>
    <ResponseField name="retry_interval" type="integer">
      Intervalo em segundos entre as tentativas de repetição
    </ResponseField>
    <ResponseField name="created_at" type="string">
      O carimbo de data/hora em que a campanha foi criada
    </ResponseField>
    <ResponseField name="updated_at" type="string">
      O carimbo de data/hora da última atualização da campanha
    </ResponseField>
  </Expandable>
</ResponseField>
