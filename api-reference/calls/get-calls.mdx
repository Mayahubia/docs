---
title: "Listar chamadas"
api: "GET /user/calls"
description: "Listar todas as chamadas do usuário autenticado com opções de filtragem
"
---

Este endpoint permite listar todas as chamadas pertencentes ao usuário autenticado com várias opções de filtragem.

### Parâmetros de consulta

<ParamField query="status" type="string">
  Filtrar chamadas por status. Valores possíveis:`initiated`, `ringing`, `busy`, `in-progress`, `ended`, `completed`, `ended_by_customer`, `ended_by_assistant`, `no-answer`, `failed`
</ParamField>

<ParamField query="type" type="string">
  Filtrar chamadas por tipo. Valores possíveis:: `inbound`, `outbound`, `web`
</ParamField>

<ParamField query="phone_number" type="string">
  Filtrar chamadas pelo número de telefone do cliente
</ParamField>

<ParamField query="assistant_id" type="integer">
  Filtrar chamadas pelo ID do assistente
</ParamField>

<ParamField query="campaign_id" type="integer">
  Filtrar chamadas pelo ID da campanha
</ParamField>

<ParamField query="date_from" type="string">
  Filtrar chamadas a partir desta data (formato YYYY-MM-DD)
</ParamField>

<ParamField query="date_to" type="string">
  Filtrar chamadas até esta data (formato YYYY-MM-DD)
</ParamField>

<ParamField query="per_page" type="integer">
  Número de chamadas por página (1-100, padrão: 15)
</ParamField>

<ParamField query="page" type="integer">
  Número da página (padrão: 1)
</ParamField>

### Campos da resposta

<ResponseField name="data" type="array">
  <Expandable title="Propriedades">
    <ResponseField name="id" type="integer">
      O identificador único da chamada
    </ResponseField>
    <ResponseField name="assistant_name" type="string">
      O nome do assistente que realizou a chamada
    </ResponseField>
    <ResponseField name="campaign_name" type="string">
      O nome da campanha à qual esta chamada pertence (se aplicável)
    </ResponseField>
    <ResponseField name="type" type="string">
      O tipo da chamada (`inbound`, `outbound`, or `web`)
    </ResponseField>
    <ResponseField name="duration" type="integer">
      A duração da chamada em segundos
    </ResponseField>
    <ResponseField name="assistant_phone_number" type="string">
      O número de telefone usado pelo assistente
    </ResponseField>
    <ResponseField name="client_phone_number" type="string">
      O número de telefone do cliente
    </ResponseField>
    <ResponseField name="status" type="string">
      O status atual da chamada
    </ResponseField>
    <ResponseField name="transcript" type="string">
      A transcrição da conversa da chamada
    </ResponseField>
    <ResponseField name="variables" type="object">
      Variáveis coletadas durante a chamada
    </ResponseField>
    <ResponseField name="evaluation" type="object">
      Dados de avaliação do desempenho da chamada
    </ResponseField>
    <ResponseField name="webhook_response" type="object">
      Resposta de quaisquer webhooks configurados
    </ResponseField>
    <ResponseField name="carrier_cost" type="number">
      O custo cobrado pela operadora por esta chamada
    </ResponseField>
    <ResponseField name="total_cost" type="number">
      O custo total da chamada, incluindo todas as taxas
    </ResponseField>
    <ResponseField name="answered_by" type="string">
      Quem atendeu a chamada (`human`, `machine`, or `unknown`)
    </ResponseField>
    <ResponseField name="recording_url" type="string">
      URL da gravação da chamada (se disponível e habilitada)
    </ResponseField>
    <ResponseField name="created_at" type="string">
      A data e hora em que a chamada foi criada
    </ResponseField>
    <ResponseField name="updated_at" type="string">
      A data e hora da última atualização da chamada
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="current_page" type="integer">
  O número da página atual
</ResponseField>

<ResponseField name="per_page" type="integer">
  Número de itens por página
</ResponseField>

<ResponseField name="total" type="integer">
  Número total de chamadas que correspondem aos critérios
</ResponseField>

<ResponseField name="last_page" type="integer">
  O número da última página
</ResponseField>
